001package us.lsi.tools;
002
003/**
004 * @author reinaqu
005 * Clase de utilidad con métodos para facilitar la gestión de restricciones.
006 */
007public class Checkers {
008
009        /**
010         * @param textoRestriccion
011         *     Mensaje que se incluirá en la excepción elevada
012         * @param condicion
013         *     Expresión booleana que ha de cumplirse para que no se eleve una excepción
014         * @throws IllegalArgumentException si no se cumple la condición
015         *  
016         */
017        public static void check(String textoRestriccion, Boolean condicion) {
018                if (!condicion) {
019                        throw new IllegalArgumentException(
020                                        Thread.currentThread().getStackTrace()[2].getClassName() +
021                                        "." + 
022                                        Thread.currentThread().getStackTrace()[2].getMethodName() +
023                                        ": " + 
024                                        textoRestriccion);
025                }
026        }
027
028        /**
029         * @param parametros Lista de parámetros para los que se quiere realizar una comprobación de nulidad.
030         * @throws IllegalArgumentException si alguno de los parámetros es nulo.
031         */
032        public static void checkNoNull(Object... parametros) {
033                for (int i = 0; i < parametros.length; i++) {
034                        if (parametros[i] == null) {
035                                throw new IllegalArgumentException(
036                                                Thread.currentThread().getStackTrace()[2].getClassName() +
037                                                "." + 
038                                                Thread.currentThread().getStackTrace()[2].getMethodName() +
039                                                ": el parámetro " + (i + 1) + " es nulo");
040                        }
041                }
042        }
043}
