001package us.lsi.tools;
002
003import java.io.BufferedReader;
004import java.io.BufferedWriter;
005import java.io.FileReader;
006import java.io.FileWriter;
007import java.io.IOException;
008import java.io.PrintWriter;
009import java.util.List;
010import java.util.stream.Collectors;
011
012public class FileTools {
013        
014        public static List<String> lineas(String file){
015                List<String> lineas = null;
016                try {
017                        BufferedReader bufferedReader = new BufferedReader(new FileReader(file));
018                        lineas = bufferedReader.lines().collect(Collectors.toList());
019                        bufferedReader.close();
020                } catch (IOException e) {
021                        System.out.println(e.toString());
022                }
023                return lineas;
024        }
025        
026        public static String text(String file){
027                List<String> lineas = null;
028                try {
029                        BufferedReader bufferedReader = new BufferedReader(new FileReader(file));
030                        lineas = bufferedReader.lines().collect(Collectors.toList());
031                        bufferedReader.close();
032                } catch (IOException e) {
033                        System.out.println(e.toString());
034                }
035                return lineas.stream().collect(Collectors.joining("\n"));
036        }
037        
038        public static void write(String file, String text){
039                try {
040                        final PrintWriter f =  new PrintWriter(new BufferedWriter(new FileWriter(file)));
041                        f.println(text);
042                        f.close();
043                } catch (IOException e) {
044                        throw new IllegalArgumentException("No se ha podido crear el fichero " + file);
045                }
046        }
047
048}
