001package us.lsi.tools;
002
003import java.text.DecimalFormat;
004import java.text.DecimalFormatSymbols;
005
006import us.lsi.data.Coordenadas2D;
007
008public class GraphicsGoogleMaps extends AbstractGaphicsMaps implements GraphicsMaps {
009        
010        public GraphicsGoogleMaps() {}
011        
012        private static Integer n = -1;
013        @Override
014        public String toMarker(String color, String text, Coordenadas2D coordenadas) {
015                String url = "\"http://maps.google.com/mapfiles/ms/icons/";
016                  url += color + "-dot.png\"";
017                DecimalFormatSymbols simbolos = new DecimalFormatSymbols();
018                simbolos.setDecimalSeparator('.');
019                DecimalFormat formateador = new DecimalFormat("####.######",simbolos);
020                String lat = formateador.format(coordenadas.getLatitude());
021                String lng = formateador.format(coordenadas.getLongitude());
022                n++;
023                return String.format("marker%d = new google.maps.Marker({\r\n" + 
024                                "    map: map,\r\n" + 
025                                "    position: {lat: %s, lng: %s},\r\n" + 
026                                "    title: '%s' ,\r\n" +
027                                "    icon: {\r\n url: " + url+ "}\r\n" + 
028                                "  });",n,lat,lng,text);
029        }
030
031        @Override
032        public String toPoint(Coordenadas2D coordenadas) {
033                DecimalFormatSymbols simbolos = new DecimalFormatSymbols();
034                simbolos.setDecimalSeparator('.');
035                DecimalFormat formateador = new DecimalFormat("####.######",simbolos);
036                String lat = formateador.format(coordenadas.getLatitude());
037                String lng = formateador.format(coordenadas.getLongitude());
038                return String.format("{lat: %s, lng: %s}",lat,lng);
039        }
040
041        @Override
042        public String getKey() {
043                return FileTools.lineas("resources/privateGoogle.txt").get(0);
044        }
045        
046
047        @Override
048        protected String getPolylinePattern() {
049                return FileTools.text("resources/GooglePolylinePattern.html");
050        }
051        
052        @Override
053        protected String getMarkersPattern() {
054                return FileTools.text("resources/GoogleMarkersPattern.html");
055        }
056}
