001package us.lsi.tools;
002
003import java.text.DecimalFormat;
004import java.text.DecimalFormatSymbols;
005
006import us.lsi.data.Coordenadas2D;
007
008public class GraphicsBingMaps extends AbstractGaphicsMaps implements GraphicsMaps {
009        
010        
011        public GraphicsBingMaps() {}
012
013        private static Integer n = -1;
014        
015        public String toPoint(Coordenadas2D coordenadas) {
016                DecimalFormatSymbols simbolos = new DecimalFormatSymbols();
017                simbolos.setDecimalSeparator('.');
018                DecimalFormat formateador = new DecimalFormat("####.######",simbolos);
019                String lat = formateador.format(coordenadas.getLatitude());
020                String lng = formateador.format(coordenadas.getLongitude());
021                return String.format("new Microsoft.Maps.Location(%s,%s)",lat,lng);
022        }
023
024        public String toMarker(String color, String text,Coordenadas2D coordenadas) {
025                DecimalFormatSymbols simbolos = new DecimalFormatSymbols();
026                simbolos.setDecimalSeparator('.');
027                DecimalFormat formateador = new DecimalFormat("####.######",simbolos);
028                String lat = formateador.format(coordenadas.getLatitude());
029                String lng = formateador.format(coordenadas.getLongitude());
030                n++;
031                return String.format(" var pin%d = new Microsoft.Maps.Pushpin(new Microsoft.Maps.Location(%s,%s), {\r\n" + 
032                                "color: '%s'," +
033                                "text: '%s'\r\n });\r\n" + 
034                                "map.entities.push(pin%d);",n,lat,lng,color,text,n);
035        }
036
037        @Override
038        public String getKey() {
039                return FileTools.lineas("resources/privateBing.txt").get(0);
040        }
041
042        @Override
043        protected String getPolylinePattern() {
044                return FileTools.text("resources/BingPolylinePattern.html");
045        }
046
047        @Override
048        protected String getMarkersPattern() {
049                return FileTools.text("resources/BingMarkersPattern.html");
050        }
051
052}
