001package us.lsi.tools;
002
003import java.util.Arrays;
004import java.util.stream.IntStream;
005
006public class Preconditions {
007
008        /**
009         * Checks that the boolean is true. Use for validating arguments to methods.
010         * @param condition A condition
011         */
012        public static void checkArgument(boolean condition){
013                if(!condition){
014                        throw new IllegalArgumentException();
015                }
016        }
017        
018        /**
019         * Checks that the boolean is true. Use for validating arguments to methods.
020         * @param message A message
021         * @param condition A condition
022         */
023        public static void checkArgument(boolean condition, String message){
024                if(!condition){
025                        throw new IllegalArgumentException(message);
026                }
027        }
028
029        /**
030         * Checks some state of the object, not dependent on the method arguments.  
031         * @param condition A condition
032         */
033        public static void checkState(boolean condition){
034                if(!condition){
035                        throw new IllegalArgumentException();
036                }
037        }
038        /**
039         * Checks some state of the object, not dependent on the method arguments. 
040         * @param message Mensaje a imprimir
041         * @param condition A condition
042         */
043        public static void checkState(boolean condition, String message){
044                if(!condition){
045                        throw new IllegalArgumentException(message);
046                }
047        }
048        
049        /**
050         * Comprueba que los parámetros son no null
051         * @param <T> El tipo del elemento      
052         * @param reference Parámetros a comprobar
053         */
054        public static <T> void checkNotNull(@SuppressWarnings("unchecked") T... reference){
055                if(Arrays.stream(reference).anyMatch(x->x == null)){
056                        throw new NullPointerException(String.format("Es nulo el elemento %d",
057                                        IntStream.range(0, reference.length)
058                                        .filter(i->reference[i]!=null)
059                                        .findFirst()
060                                        .getAsInt()));
061                }
062        }
063                
064        /**
065         * Checks that index is a valid element index into a list, string, or array with the specified size. 
066         * An element index may range from 0 inclusive to size exclusive. 
067         * You don't pass the list, string, or array directly; you just pass its size. 
068         * @param index Un índice 
069         * @param size El tamaño de la lista
070         * @return Index El índice del elemento
071         */
072        public static int checkElementIndex(int index, int size){
073                if(!(index>=0 && index<size)){
074                        throw new IndexOutOfBoundsException(String.format("Index = %d, size %d", index,size));
075                }
076                return index;
077        }
078        
079        /**
080         * Checks that index is a valid position index into a list, string, or array with the specified size. 
081         * A position index may range from 0 inclusive to size inclusive. 
082         * You don't pass the list, string, or array directly; you just pass its size. Returns index.
083         * @param index El índice del elemento
084         * @param size El tamaño de la lista
085         * @return Index El índice del elemento
086         */
087        public static int checkPositionIndex(int index, int size){
088                if(!(index>=0 && index<=size)){
089                        throw new IndexOutOfBoundsException(String.format("Index = %d, size %d", index,size));
090                }
091                return index;
092        }
093        
094        
095}
