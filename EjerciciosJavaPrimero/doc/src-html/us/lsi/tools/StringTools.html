001package us.lsi.tools;
002
003import java.util.Map;
004import java.util.regex.Pattern;
005
006public class StringTools {
007        
008        /**
009         * @param in String de entrada
010         * @param reglas Conjunto de reglas compuestas por un iodentificador y un texto asociado
011         * @return Devuelve le String de entrada donde se ha sustituido {nombre} por el texto asociado a nombre en las reglas
012         */
013        public static String transform(String in, Map<String,String> reglas) {
014                String out = in;
015                Pattern pattern;
016                for(String p:reglas.keySet()) {
017                        pattern = Pattern.compile("\\{"+p+"\\}");
018                        out = pattern.matcher(out).replaceAll(reglas.get(p));
019                }
020                return out;
021        }
022        
023        
024}
