001package us.lsi.tools;
002
003public class MutableType<T> {
004        
005        public static <T> MutableType<T> of(T e) {
006                return new MutableType<T>(e);
007        }
008        
009        public T value;
010        
011        public T newValue(T newValue) {
012                T old = value;
013                this.value = newValue;
014                return old;
015        }
016        
017        private MutableType(T e) {
018                super();
019                this.value = e;
020        }
021        @Override
022        public String toString() {
023                return value.toString();
024        }
025        
026
027}
