001package us.lsi.tools;
002
003import java.util.function.Function;
004
005public class Operators {
006        
007        /**
008         * @param <E> Tipo de los elementos
009         * @param <R> Tipo del resultado
010         * @param e Un elemento que puede ser null
011         * @param value El valor por defecto que se delverá si e es null
012         * @param expression La expresióbn que se aplicará a e si no es null
013         * @return Si e es null se devolverçá value y si no lo es expression.aply(e)
014         */
015        public static <E,R> R ifNull(E e, R value, Function<E,R> expression) {
016                return e==null? value: expression.apply(e);
017        }
018        /**
019         * @param <E> Tipo de los elementos
020         * @param e Un elemento que puede ser null
021         * @param value El valor por defecto que se delverá si e es null
022         * @return Si e es null se devolverçá value y si no lo es e
023         */
024        public static <E> E ifNull(E e, E value) {
025                return e==null? value: e;
026        }
027
028}
