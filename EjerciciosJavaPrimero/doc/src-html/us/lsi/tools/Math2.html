001package us.lsi.tools;
002
003import java.util.Random;
004
005public class Math2 {
006        
007        private static Random random = new Random(System.nanoTime());
008        
009        /**
010         * @param limitExcluded Un entero
011         * @return Un entero aleatorio en el intervalo 0..limitExcluded con el extremo derecho excluido
012         */
013        public static Integer intRandom(Integer limitExcluded) {        
014                Double r = random.nextDouble();
015                Integer ri = (int) (r*limitExcluded);
016                return ri;
017        }
018
019}
