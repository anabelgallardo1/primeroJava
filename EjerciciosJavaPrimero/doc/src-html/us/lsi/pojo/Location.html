001package us.lsi.pojo;
002
003import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
004
005@JsonIgnoreProperties(ignoreUnknown = true)
006public class Location {
007        private String country;
008
009        private String city;
010
011        private String latitude;
012
013        private String longitude;
014
015        public String getCountry() {
016                return country;
017        }
018
019        public Location() {
020                super();
021        }
022
023
024        public void setCountry(String country) {
025                this.country = country;
026        }
027
028        public String getCity() {
029                return city;
030        }
031
032        public void setCity(String city) {
033                this.city = city;
034        }
035
036        public String getLatitude() {
037                return latitude;
038        }
039
040        public void setLatitude(String latitude) {
041                this.latitude = latitude;
042        }
043
044        public String getLongitude() {
045                return longitude;
046        }
047
048        public void setLongitude(String longitude) {
049                this.longitude = longitude;
050        }
051
052        @Override
053        public String toString() {
054                return "Location [country = " + country + ", city = " + city + ", latitude = " + latitude + ", longitude = "
055                                + longitude + "]";
056        }
057}
