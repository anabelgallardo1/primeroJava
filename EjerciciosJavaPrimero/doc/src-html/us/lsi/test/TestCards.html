001package us.lsi.test;
002
003import java.util.Arrays;
004import java.util.Comparator;
005import java.util.List;
006import java.util.Map;
007import java.util.stream.Collectors;
008import java.util.stream.IntStream;
009
010import us.lsi.montecarlo.Mano;
011import us.lsi.montecarlo.Mano.Jugada;
012import us.lsi.tools.Graphics;
013import us.lsi.tools.GraphicsCards;
014
015public class TestCards {
016        
017        public static Map<Jugada, Long> random(Integer n){
018                Map<Jugada,Long> m = IntStream.range(0,n)
019                                   .boxed()
020                                   .map(x->Mano.random())
021                                   .collect(Collectors.groupingBy(x->x.getJugada(),Collectors.counting()));
022                return m;
023        }
024
025        public static void main(String[] args) {
026                Mano m = Mano.random();
027                Double fuerza = Mano.fuerza(m, 5000);
028                GraphicsCards.cartas("ficheros/CartasOut.html",m.getCartas(),fuerza,m.getJugada().toString());
029                Map<Jugada,Long> tj = TestCards.random(5000);
030                System.out.println(tj.entrySet().stream()
031                                .sorted(Comparator.comparing(x->x.getValue()))
032                                .map(x->String.format("(%s,%d)",x.getKey(),x.getValue()))
033                                .collect(Collectors.joining("\n")));
034                List<String> campos = Arrays.asList("Tipo de Jugada","Numero");
035                List<String> nombres = tj.keySet().stream().map(x->x.toString()).collect(Collectors.toList());
036                List<Long> datos = tj.keySet().stream().map(x->tj.get(x)).collect(Collectors.toList());
037                Graphics.pieChart("ficheros/PieChartOut.html","Tipos de jugadas",campos, nombres, datos);
038        }
039
040        
041
042}
